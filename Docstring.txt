class Axis:
    """
    通用坐标轴类，用于生成和管理一维均匀采样坐标轴数据
    
    Attributes
    ----------
    N : int
        坐标轴数据点数
    dx : float
        坐标轴采样间隔
    x0 : float
        坐标轴起始点
    name : str
        坐标轴数据名称
    unit : str
        坐标轴数据单位
    data : np.ndarray
        坐标轴数据数组
    lim : tuple
        坐标轴数据范围 (min, max)
    label : str
        坐标轴标签 (name/unit)
    
    Methods
    -------
    copy() -> Axis
        返回坐标轴对象的深拷贝
    """
